<script>
const verbs = [
  {inf:"machen", du:"mach", ihr:"macht", sie:"machen Sie"},
  {inf:"gehen", du:"geh", ihr:"geht", sie:"gehen Sie"},
  {inf:"lesen", du:"lies", ihr:"lest", sie:"lesen Sie"},
  {inf:"öffnen", du:"öffne", ihr:"öffnet", sie:"öffnen Sie"},
  {inf:"holen", du:"hol", ihr:"holt", sie:"holen Sie"}
];

let correct = "";

function newTask() {
  const v = verbs[Math.floor(Math.random() * verbs.length)];
  const s = ["du","ihr","Sie"][Math.floor(Math.random() * 3)];

  document.getElementById("sentence").innerText =
    s.toUpperCase() + ": " + v.inf + " das Buch!";

  if (s === "du") correct = v.du;
  else if (s === "ihr") correct = v.ihr;
  else correct = v.sie;

  document.getElementById("answer").value = "";
  document.getElementById("result").innerText = "";
}

function check() {
  const u = document.getElementById("answer").value.trim().toLowerCase();

  if (u === correct.toLowerCase()) {
    document.getElementById("result").innerText = "✔ Richtig!";
    document.getElementById("result").style.color = "green";
  } else {
    document.getElementById("result").innerText = "✘ Falsch → " + correct;
    document.getElementById("result").style.color = "red";
  }
}

newTask();
</script>
